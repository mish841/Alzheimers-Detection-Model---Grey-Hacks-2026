PYTHON · MODEL SETUP 
# Load pretrained ResNet18 and adapt for 2-class output
import torch
import torch.nn as nn
from torchvision import models

# Initialise ResNet18 with ImageNet weights
model = models.resnet18(weights=models.ResNet18_Weights.DEFAULT)

# Replace final fully-connected layer → 2 classes
# (Healthy vs Alzheimer's)
model.fc = nn.Linear(model.fc.in_features, 2)

device = torch.device("cuda" if torch.cuda.is_available() else "cpu")
model = model.to(device)

# Freeze CNN layers, unfreeze classifier head only
for param in model.parameters():
    param.requires_grad = False

for param in model.fc.parameters():
    param.requires_grad = True

# Load trained weights
model.load_state_dict(torch.load("alz_model.pth", map_location=device))
model.eval()
# Image preprocessing pipeline for ResNet18 input
from torchvision import transforms
from PIL import Image

# ResNet18 expects 224×224 with ImageNet normalisation
transform = transforms.Compose([
    transforms.Resize((224, 224)),
    transforms.ToTensor(),
    transforms.Normalize(
        mean=[0.485, 0.456, 0.406],  # ImageNet mean
        std= [0.229, 0.224, 0.225]   # ImageNet std
    )
])

# Load and convert MRI to RGB (handles grayscale DICOMs)
image = Image.open(image_path).convert("RGB")

# Add batch dimension: [1, 3, 224, 224]
input_tensor = transform(image).unsqueeze(0).to(device)
# Run inference and get class probabilities
import torch

classes = ["Healthy", "Alzheimer's"]

with torch.no_grad():
    output = model(input_tensor)

# Convert logits → probabilities via softmax
probs = torch.softmax(output, dim=1)
prediction = torch.argmax(probs, dim=1).item()
confidence = probs[0][prediction].item()

print(f"Prediction : {classes[prediction]}")
print(f"Confidence : {confidence:.2%}")
print(f"Healthy    : {probs[0][0].item():.2%}")
print(f"Alzheimer's: {probs[0][1].item():.2%}")
# Grad-CAM: visualise which regions drove the prediction
from pytorch_grad_cam import GradCAM
from pytorch_grad_cam.utils.image import show_cam_on_image
import numpy as np
import matplotlib.pyplot as plt

# Hook into the last convolutional block of ResNet18
target_layer = model.layer4[-1]

cam = GradCAM(model=model, target_layers=[target_layer])
grayscale_cam = cam(input_tensor=input_tensor)[0]

# Normalise original image to [0, 1] for overlay
rgb_img = np.array(image.resize((224, 224))) / 255.0
visualization = show_cam_on_image(rgb_img, grayscale_cam, use_rgb=True)

plt.imshow(visualization)
plt.title("Grad-CAM Attention Map")
plt.axis("off")
plt.show()
